            \State $ sell \gets 0 $            
            \For{i from 1 to n + 1}
                \If{B[buy] > B[i]}
                    \State $ buy \gets i $                    
                    \State $ sell \gets i $
                \Else
                    \If{S[sell] < S[i]}
                        \State $ sell \gets i $
                    \EndIf                     
                \EndIf                                
            \EndFor            
            \\
            \State $ \Return \text{ (buy, sell)} $
